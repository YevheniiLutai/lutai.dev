const checkboxes = document.querySelectorAll('input[name="group1"]');
const aboutBlocks = document.querySelectorAll('.about__block');

function displayBlock(index) {
  checkboxes.forEach((checkbox, i) => {
    if (i !== index) {
      checkbox.checked = false;
      aboutBlocks[i].style.display = 'none';
    }
  });

  const checkbox = checkboxes[index];
  const aboutBlock = aboutBlocks[index];

  checkbox.addEventListener('change', () => {
    aboutBlock.style.display = checkbox.checked ? 'block' : 'none';
  });
}

aboutBlocks.forEach((_, i) => {
  displayBlock(i);
});
